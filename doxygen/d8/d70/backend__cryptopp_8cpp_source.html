27EcdsaSignature BackendCryptoPP::sign_data(const PrivateKey& private_key, const ByteBuffer& data)
28{
29 // calculate signature
30 Signer signer(private_key);
31 ByteBuffer signature(signer.MaxSignatureLength(), 0x00);
32 auto signature_length = signer.SignMessage(m_prng, data.data(), data.size(), signature.data());
33 signature.resize(signature_length);
34
35 auto signature_delimiter = signature.begin();
36 std::advance(signature_delimiter, 32);
37
38 EcdsaSignature ecdsa_signature;
39 // set R
40 X_Coordinate_Only coordinate;
41 coordinate.x = ByteBuffer(signature.begin(), signature_delimiter);
42 ecdsa_signature.R = std::move(coordinate);
43 // set s
44 ByteBuffer trailer_field_buffer(signature_delimiter, signature.end());
45 ecdsa_signature.s = std::move(trailer_field_buffer);
46
47 return ecdsa_signature;
48}

63boost::optional<Uncompressed> BackendCryptoPP::decompress_point(const EccPoint& ecc_point)
64{
65 struct DecompressionVisitor : public boost::static_visitor<bool>
66 {
67 bool operator()(const X_Coordinate_Only&)
68 {
69 return false;
70 }
71
72 bool operator()(const Compressed_Lsb_Y_0& p)
73 {
74 decompress(p.x, 0x02);
75 return true;
76 }
77
78 bool operator()(const Compressed_Lsb_Y_1& p)
79 {
80 decompress(p.x, 0x03);
81 return true;
82 }
83
84 bool operator()(const Uncompressed& p)
85 {
86 result = p;
87 return true;
88 }
89
90 void decompress(const ByteBuffer& x, ByteBuffer::value_type type)
91 {
92 ByteBuffer compact;
93 compact.reserve(x.size() + 1);
94 compact.push_back(type);
95 std::copy(x.begin(), x.end(), std::back_inserter(compact));
96
97 BackendCryptoPP::Point point;
98 CryptoPP::DL_GroupParameters_EC<CryptoPP::ECP> group(CryptoPP::ASN1::secp256r1());
99 group.GetCurve().DecodePoint(point, compact.data(), compact.size());
100
101 result.x = x;
102 result.y.resize(result.x.size());
103 point.y.Encode(result.y.data(), result.y.size());
104 }
105
106 Uncompressed result;
107 };
108
109 DecompressionVisitor visitor;
110 if (boost::apply_visitor(visitor, ecc_point)) {
111 return visitor.result;
112 } else {
113 return boost::none;
114 }
115}

117ecdsa256::KeyPair BackendCryptoPP::generate_key_pair()
118{
119 ecdsa256::KeyPair kp;
120 auto private_key = generate_private_key();
121 auto& private_exponent = private_key.GetPrivateExponent();
122 assert(kp.private_key.key.size() >= private_exponent.ByteCount());
123 private_exponent.Encode(kp.private_key.key.data(), kp.private_key.key.size());
124
125 auto public_key = generate_public_key(private_key);
126 auto& public_element = public_key.GetPublicElement();
127 assert(kp.public_key.x.size() >= public_element.x.ByteCount());
128 assert(kp.public_key.y.size() >= public_element.y.ByteCount());
129 public_element.x.Encode(kp.public_key.x.data(), kp.public_key.x.size());
130 public_element.y.Encode(kp.public_key.y.data(), kp.public_key.y.size());
131 return kp;
132}
